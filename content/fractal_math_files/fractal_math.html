function tree(n,angle)
 
numBranch=n+1;
initialLength=1;
 
line([0 0],[-initialLength,0])
pause(1.5)
axis on
Branches(90,[0,0], numBranch-1,angle, numBranch,initialLength);
function Branches(initialAngle,point,iterations,rotation,totalIterations,oldLength)
length=oldLength*0.5;
x1=point(1); y1=point(2);
angleL=initialAngle+rotation; % rotate left angle
angleR=initialAngle-rotation; % rotate right angle
x2= length*cosd(angleL)+x1; y2= length*sind(angleL)+y1;
x3= length*cosd(angleR)+x1; y3= length*sind(angleR)+y1;
line([x1,x2],[y1,y2]);
pause(1.5)
line([x1,x3],[y1,y3]);
pause(1.5)
if iterations-1>0
Branches(angleL,[x2,y2],iterations-1,rotation,totalIterations,length);
Branches(angleR,[x3,y3],iterations-1,rotation,totalIterations,length);
end
end
end
